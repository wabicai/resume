let getObj = { "a.b.c": 1, "a.d": 2, e: 3 };
function getObj(obj) {
  let newObj = {};
  function helper(keys, value, newObj) {
    const key = keys.shift();
    if (keys.length === 0) {
      newObj[key] = value;
    } else {
      if (!newObj[key]) {
        newObj[key] = {};
      }
      helper(keys, value, newObj[key]);
    }
  }
  for (let index in obj) {
    let currKeys = index.split(".");
    let value = obj[index];
