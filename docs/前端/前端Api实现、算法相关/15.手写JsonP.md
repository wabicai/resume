function request(options) {
  return new Promise((resolve, reject) => {
    const { url, timeout } = options;
    let requestNode = document.createElement("script");
    //创建请求
    requestNode.src = url;
    // <script src="">
    Promise.race([
      setTimeout(function () {
        reject("没有返回数据");
      }, timeout),
      (window.cb = (result) => {
        resolve(result);
      }),
    ]);

}
