// 使用哈希表版
function listToTreeSimple(data) {
  const res = [];
  const obj = {};
  data.forEach((item) => (obj[item.id] = item));
  data.forEach((item) => {
    const parent = obj[item.parentId];
    if (parent) {
      parent.children = parent.children || [];
      parent.children.push(item);
    } else {
      // * 根节点
      res.push(item);
    }
  });
  console.log(res);
}
